```python
class UserService:
    @staticmethod
    async def create(db: AsyncSession, user_data: UserCreate):
        # Validar email Ãºnico
        existing = await UserService.get_by_email(db, user_data.email)
        if existing:
            raise ConflictException("Email ya existe")
        
        # Crear usuario
        hashed_password = get_password_hash(user_data.password)
        db_user = User(
            email=user_data.email,
            hashed_password=hashed_password
        )
        db.add(db_user)
        await db.commit()
        return db_user
```
